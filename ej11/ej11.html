Ejercicio 11

Escribe un programa que genere 100 números del 1 al 20. Debe mostrar la cuenta de los números que al menos han aparecido una vez.

const arr= [];

function getRandom(min,max){

return Math.floor(Math.random() * (Number(max) - Number(min) + 1)) + Number(min);

}

for(let i = 0; i < 100; i++) arr.unshift(getRandom(1,20));

console.log(arr);

let unicos = new Set(arr);

console.log(unicos);

let contador = 0;

for(let unico of unicos){

for(let i = 0;i < arr.length; i++){

if(unico == arr[i]) contador++;

}

console.log(`número: ${unico} | contador: ${contador}`);

contador = 0;

}
