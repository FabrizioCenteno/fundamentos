Ejercicio 15

Escribe un programa que imprima todos los números primos entre 1 y un número dado por el usuario.

const max = prompt('número máximo');

function esPrimo(maximo){

for(let i = 1; i <= maximo; i++) {

let primo = true;

if(i < 2){

primo = false;

}else{;

for(let j = 2; j <= Math.sqrt(i); j++){

if(i % j === 0){

primo = false;

}

}

}

console.log(`el número ${i} ${primo? 'es primo':'no es primo'}`);

}

}

esPrimo(max);
